name: Pull Request Validation

on:
  pull_request:
    branches:
      - dev
      - dev/prod
      - main

jobs:
  validate:
    name: Validate Infrastructure
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'
          
      - name: Set up Terraform
        uses: hashicorp/setup-terraform@v2
        with:
          terraform_version: '1.5.0'
          
      - name: Set up Helm
        uses: azure/setup-helm@v3
        with:
          version: 'v3.12.0'
          
      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install pytest pylint yamllint kubernetes-validate
          pip install -r requirements.txt
          
      - name: Lint Python code
        run: |
          pylint scripts/ tests/ --disable=C0111,C0103

      - name: Validate Helm charts
        run: |
          chmod +x ./scripts/validate_helm.sh
          ./scripts/validate_helm.sh
          
      - name: Terraform format check
        run: terraform fmt -check -recursive
        
      - name: Terraform validation
        run: |
          chmod +x ./scripts/validate_terraform.sh
          ./scripts/validate_terraform.sh
          
      - name: Run security scan
        run: |
          chmod +x ./scripts/security_scan.sh
          ./scripts/security_scan.sh
          
      - name: Run unit tests
        run: |
          pytest tests/unit/

      - name: Run integration tests
        run: |
          pytest tests/integration/
          
      - name: Deployment validation
        run: |
          python ./scripts/deployment_validation.py
          
      - name: Summarize findings
        if: always()
        run: |
          echo "## Validation Results" >> $GITHUB_STEP_SUMMARY
          echo "- Helm: ${{ job.steps.validateHelm.outcome }}" >> $GITHUB_STEP_SUMMARY
          echo "- Terraform: ${{ job.steps.validateTerraform.outcome }}" >> $GITHUB_STEP_SUMMARY
          echo "- Security: ${{ job.steps.securityScan.outcome }}" >> $GITHUB_STEP_SUMMARY
          echo "- Tests: ${{ job.steps.unitTests.outcome }}" >> $GITHUB_STEP_SUMMARY
